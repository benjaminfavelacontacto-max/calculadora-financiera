<script>

let chart;

let resultadoGlobal = {};

function formatoMoneda(valor){

    return "$" + Number(valor).toLocaleString('es-MX', {minimumFractionDigits:2, maximumFractionDigits:2});

}

function calcular() {

    const capital = parseFloat(document.getElementById("capital").value) || 0;

    const aportacion = parseFloat(document.getElementById("aportacion").value) || 0;

    const tasaAnual = parseFloat(document.getElementById("tasa").value) / 100 || 0;

    const años = parseInt(document.getElementById("años").value) || 0;

    const frecuencia = parseInt(document.getElementById("frecuencia").value);

    if (capital <= 0 || tasaAnual <= 0 || años <= 0) {

        alert("Por favor, ingresa valores válidos para capital, tasa y años.");

        return;

    }

    let labels = [];

    let simpleData = [];

    let compuestoData = [];

   

    let simple = capital;

    let compuesto = capital;

    let totalInvertido = capital;

    // Cálculo año tras año para la gráfica

    for (let i = 0; i <= años; i++) {

        labels.push("Año " + i);

       

        if (i === 0) {

            simpleData.push(capital);

            compuestoData.push(capital);

        } else {

            // Interés Simple (sobre el capital inicial + aportaciones sin reinvertir intereses)

            totalInvertido += (aportacion * 12);

            simple = capital + (capital * tasaAnual * i) + (aportacion * 12 * i);

           

            // Interés Compuesto

            for(let m = 0; m < 12; m++) {

                compuesto = (compuesto + aportacion) * (1 + (tasaAnual / 12));

            }

           

            simpleData.push(simple);

            compuestoData.push(compuesto);

        }

    }

    resultadoGlobal = {

        simple,

        compuesto,

        invertido: totalInvertido,

        diferencia: compuesto - simple

    };

    const resumenDiv = document.getElementById("resumen");

    resumenDiv.style.display = "block";

    resumenDiv.innerHTML = `

        <div>Total Invertido (Capital + Aportes): <span>${formatoMoneda(totalInvertido)}</span></div>

        <div>Final con Interés Simple: <span>${formatoMoneda(simple)}</span></div>

        <div style="color: #d4af37;">Final con Interés Compuesto: <span>${formatoMoneda(compuesto)}</span></div>

        <hr style="border: 0.5px solid #333; margin: 10px 0;">

        <div style="font-weight: bold;">Ganancia Extra por Interés Compuesto: <span>${formatoMoneda(resultadoGlobal.diferencia)}</span></div>

    `;

    actualizarGrafica(labels, simpleData, compuestoData);

}

function actualizarGrafica(labels, simpleData, compuestoData) {

    if (chart) chart.destroy();

    const ctx = document.getElementById("grafica").getContext("2d");

    chart = new Chart(ctx, {

        type: 'line',

        data: {

            labels: labels,

            datasets: [

                {

                    label: 'Interés Compuesto',

                    data: compuestoData,

                    borderColor: '#d4af37',

                    backgroundColor: 'rgba(212, 175, 55, 0.1)',

                    fill: true,

                    borderWidth: 3,

                    tension: 0.4

                },

                {

                    label: 'Interés Simple',

                    data: simpleData,

                    borderColor: '#444',

                    borderDash: [5, 5],

                    fill: false,

                    borderWidth: 2

                }

            ]

        },

        options: {

            responsive: true,

            plugins: {

                legend: { labels: { color: '#f5d27a', font: { size: 12 } } }

            },

            scales: {

                y: { ticks: { color: '#aaa', callback: v => '$' + v.toLocaleString() }, grid: { color: '#222' } },

                x: { ticks: { color: '#aaa' }, grid: { display: false } }

            }

        }

    });

}

function descargarPDF() {

    if (!resultadoGlobal.compuesto) {

        alert("Primero realiza un cálculo.");

        return;

    }

    const { jsPDF } = window.jspdf;

    const doc = new jsPDF();

   

    doc.setFont("helvetica", "bold");

    doc.text("REPORTE DE PROYECCIÓN FINANCIERA", 20, 20);

    doc.setFont("helvetica", "normal");

    doc.setFontSize(12);

    doc.text(`Consultor: Benjamin Favela`, 20, 30);

    doc.line(20, 35, 190, 35);

    doc.text(`Total invertido: ${formatoMoneda(resultadoGlobal.invertido)}`, 20, 50);

    doc.text(`Resultado Interés Simple: ${formatoMoneda(resultadoGlobal.simple)}`, 20, 60);

    doc.text(`Resultado Interés Compuesto: ${formatoMoneda(resultadoGlobal.compuesto)}`, 20, 70);

   

    doc.setFont("helvetica", "bold");

    doc.setTextColor(184, 150, 46);

    doc.text(`Diferencia a favor: ${formatoMoneda(resultadoGlobal.diferencia)}`, 20, 85);

    doc.save("Reporte_Inversion_Benjamin_Favela.pdf");

}
